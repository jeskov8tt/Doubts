\begin{enumerate}[$\mathcal{D}$1.]
    \item Evaluate $$\displaystyle\int \left\{\lfloor x \rfloor \right\} \, dx $$
    \item Evaluate $$\displaystyle\int \dfrac{x^2+1}{2x^4-x^2+1}  \, dx $$
    \item Evaluate $$\displaystyle\int \dfrac{-x^3}{\sqrt{15x^2+7x+1}}  \, dx $$
    \item Evaluate $$\displaystyle\int (x-1)^101 \cdot x^26 \, dx $$
    \item Evaluate $$\displaystyle\int x^{-\sfrac{2}{3}}(1+x^{\sfrac{2}{3}})^-1 \, dx $$
    \item Evaluate $$\displaystyle\int \ln (\sin x) \, dx $$
    \item Evaluate $$\displaystyle\int \dfrac{1}{x+\sqrt{x^2-x+1}}  \, dx $$
    \item Evaluate $$\displaystyle\flexibleint {\dfrac{(\tan x)^{\sfrac{-7}{6}} - (\tan x )^{\sfrac{-17}{6}}}{\sqrt[3]{\tan x} \sqrt{\sec^2 x + \tan x}+ \sqrt{\tan x} \sqrt[3]{\sec^2 x + \tan x}}  \, dx}$$
    \item If $y(x-y)^2=x$\\ then Evaluate, $$\displaystyle\int \dfrac{1}{x-3y}  \, dx $$
    \item Evaluate $$\flexibleint {\dfrac{\sin x \left(\cos^2 x - \cos^2 \dfrac{\pi}{5} \right)\left(\cos^2 x - \cos^2 \dfrac{2\pi}{5}\right)}{\sin 5x}  \, dx }$$
    \item Evaluate $$\displaystyle\int \dfrac{\sin x - x \cos x}{x^2}  \, dx $$
    \item Evaluate $$\displaystyle\flexibleint{ \left(\prod_{r=1}^{n} x+r\right)\left(\nsum[1.4]_{k=1}^{n} \dfrac{1}{x+k} \right) \, dx }$$
    \item Evaluate $$\lim\limits_{n \to \infty}{\displaystyle\nsum_{r=0}^{n}\dfrac{ \displaystyle{n\choose r} }{n^r(r+3)} }$$
\end{enumerate}
